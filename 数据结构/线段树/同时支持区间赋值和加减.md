emplate <typename T>
class segment_tree
{
private:
    vector<T> tree;
    vector<T> __add, __set;
    T zero;
    vector<bool> setflag;
    inline void down(int k, int len)
    {
        if (setflag[k])
        {
            __add[k << 1] = zero;
            __add[k << 1 | 1] = zero;
            __set[k << 1] = __set[k];
            __set[k << 1 | 1] = __set[k];
            tree[k << 1] = __set[k] * (len - (len >> 1));
            tree[k << 1 | 1] = __set[k] * (len >> 1);
            setflag[k << 1] = true;
            setflag[k << 1 | 1] = true;
            setflag[k] = false;
        }
        if (__add[k] != zero)
        {
            __add[k << 1] += __add[k];
            __add[k << 1 | 1] += __add[k];
            tree[k << 1] += __add[k] * (len - (len >> 1));
            tree[k << 1 | 1] += __add[k] * (len >> 1);
            __add[k] = zero;
        }
    }

public:
    int size;
    segment_tree(int n) : size(n), zero(T{})
    {
        tree.resize(4 * n + 5);
        __add.resize(4 * n + 5);
        __set.resize(4 * n + 5);
        setflag.resize(4 * n + 5, false);
    }
    void build(const T *src = nullptr, int ll = 1, int rr = -1, int k = 1)
    {
        if (rr == -1)
            rr = size;
        if (ll == rr)
        {
            if (src == nullptr)
                cin >> tree[k];
            else
                tree[k] = src[k];
            return;
        }
        int mid = (ll + rr) >> 1;
        build(src, ll, mid, k << 1);
        build(src, mid + 1, rr, k << 1 | 1);
        tree[k] = tree[k << 1] + tree[k << 1 | 1];
    }
    void set(const int l, const int r, const T value, int ll = 1, int rr = -1, int k = 1)
    {
        if (rr == -1)
            rr = size;
        if (ll >= l && rr <= r)
        {
            tree[k] = value * (rr - ll + 1);
            __add[k] = zero;
            __set[k] = value;
            setflag[k] = true;
            return;
        }
        down(k, rr - ll + 1);
        int mid = (ll + rr) >> 1;
        if (r > mid)
            set(l, r, value, mid + 1, rr, k << 1 | 1);
        if (l <= mid)
            set(l, r, value, ll, mid, k << 1);
        tree[k] = tree[k << 1] + tree[k << 1 | 1];
    }
    void modify(const int l, const int r, const T value, int ll = 1, int rr = -1, int k = 1)
    {
        if (rr == -1)
            rr = size;
        if (ll >= l && rr <= r)
        {
            tree[k] += value * (rr - ll + 1);
            __add[k] += value;
            return;
        }
        down(k, rr - ll + 1);
        int mid = (ll + rr) >> 1;
        if (r > mid)
            modify(l, r, value, mid + 1, rr, k << 1 | 1);
        if (l <= mid)
            modify(l, r, value, ll, mid, k << 1);
        tree[k] = tree[k << 1] + tree[k << 1 | 1];
    }
    T query(const int l, const int r, int ll = 1, int rr = -1, int k = 1)
    {
        if (rr == -1)
            rr = size;
        if (ll >= l && rr <= r)
            return tree[k];
        down(k, rr - ll + 1);
        int mid = (ll + rr) >> 1;
        T ans{};
        if (r > mid)
            ans += query(l, r, mid + 1, rr, k << 1 | 1);
        if (l <= mid)
            ans += query(l, r, ll, mid, k << 1);
        return ans;
    }
};
