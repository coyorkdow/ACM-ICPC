线段树模板（求区间和）
```c++
template<typename T>
class segment_tree {
private:
	vector<T>tree;
	vector<T>lazy;
	inline void down(int k, int len) {
		lazy[k << 1] += lazy[k];
		lazy[k << 1 | 1] += lazy[k];
		tree[k << 1] += lazy[k] * (len - (len >> 1));
		tree[k << 1 | 1] += lazy[k] * (len >> 1);
		lazy[k] = 0;
	}
public:
	int size;
	segment_tree(int n) :size(n) {
		tree.resize(4 * n + 5);
		lazy.resize(4 * n + 5);
	}
	void build(const T* src = nullptr, int __l = 1, int __r = -1, int k = 1) {
		if (__r == -1)
			__r = size;
		if (__l == __r) {
			if (src == nullptr)
				cin >> tree[k];
			else
				tree[k] = src[k];
			return;
		}
		int mid = (__l + __r) >> 1;
		build(src, __l, mid, k << 1);
		build(src, mid + 1, __r, k << 1 | 1);
		tree[k] = tree[k << 1] + tree[k << 1 | 1];
	}
	void modify(const int l, const int r, const T value, int __l = 1, int __r = -1, int k = 1) {
		if (__r == -1)
			__r = size;
		if (__l >= l && __r <= r) {
			tree[k] += value * (__r - __l + 1);
			lazy[k] += value;
			return;
		}
		if (lazy[k] != 0)
			down(k, __r - __l + 1);
		int mid = (__l + __r) >> 1;
		if (r > mid)
			modify(l, r, value, mid + 1, __r, k << 1 | 1);
		if (l <= mid)
			modify(l, r, value, __l, mid, k << 1);
		tree[k] = tree[k << 1] + tree[k << 1 | 1];
	}
	T query(const int l, const int r, int __l = 1, int __r = -1, int k = 1) {
		if (__r == -1)
			__r = size;
		if (__l >= l && __r <= r)
			return tree[k];
		if (lazy[k] != 0)
			down(k, __r - __l + 1);
		int mid = (__l + __r) >> 1;
		T ans{};
		if (r > mid)
			ans += query(l, r, mid + 1, __r, k << 1 | 1);
		if (l <= mid)
			ans += query(l, r, __l, mid, k << 1);
		return ans;
	}
};
```
